### 一、实现文件压缩
+ 根据文件地址，获取blob格式文件流
+ 通过`jszip`将 blob文件流 处理成 zip格式blob文件流
+ 通过`file-saver`将 zip格式文件流 保存在本地磁盘

```javascript
import JSZip from 'jszip';
import FileSaver from 'file-saver';
import axios from 'axios';

// 根据文件路径获取文件
const getFile = url => {
  return new Promise((resolve, reject) => {
    axios({
      methods: 'get',
      url,
      // responseType: 'arraybuffer'
      responseType: 'blob'
    })
      .then(data => {
        resolve(data.data);
      })
      .catch(error => {
        reject(error.toString());
      });
  });
};

export default function (fileList) {
  // 输出压缩时间
  console.time('exportAllFileTime');
  // 输入的文件是链接地址
  // fileList eg: [{url: 'xxx.pdf', fileName: '附件'}];

  // 通过jszip将 blob文件流 处理成 zip格式blob文件流
  const zip = new JSZip();
  const cache = {};
  const promises = [];

  this.isDowning = true;

  fileList.forEach(item => {
    const { url, fileName } = item;
    const promise = getFile(url).then(data => {
      // 逐个添加文件
      zip.file(fileName, data, { binary: true });
      cache[fileName] = data;
      console.log(cache);
    });
    promises.push(promise);
  });

  // 通过file-saver将 zip格式文件流 保存在本地磁盘
  Promise.all(promises).then(() => {
    // 生成二进制流
    zip.generateAsync({ type: 'blob' }).then(content => {
      FileSaver.saveAs(content, '文件下载.zip');
      console.timeEnd('exportAllFileTime');
      this.isDowning = false;
    });
  });
}
```
