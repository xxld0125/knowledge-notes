# 需求背景
所谓呼叫中心400，就是PC端的在线接听拨打电话的功能。此功能是由第三方厂商提供SDK，客服业务负责接入。目前有很多开发商的客服服务都有接入此功能。所以对客服业务来讲它是一个常见的功能。但是由于移动客服呼叫中心每次接入<font style="color:#DF2A3F;">新</font>400厂商时，需要投入预计 **<font style="color:#DF2A3F;">10人/天</font>** 进行对接，成本较高，而且对接过程也相对成熟，所以决定将此功能交由svip团队对接。

```html
  //  1：座席状态发生变化，strEventData=工号， strEventInfo=工作状态

  //  6：有用户呼入信息，strEventData=分机号， strEventInfo=

  function myATClient_onReceiveCTI_EventChanged(
    EventCode,
    EventData,
    EventInfo
  ) {
    var strStatus;
    switch (EventCode) {
      case 1: //1：表示座席员状态发生变化，EventData 表示座席员工号
        strStatus = "";
        show_agent_status(document, EventData, strStatus);
        break;
      case 2: //表示座席员呼叫状态发生变化，EventData 表示座席员工号
        show_agent_status(document, EventData, strStatus);

        var callinfo = ATGetCallInfo(EventData, login_ext, "", 1); //获取电话状态
        console.log(
          "111:" +
            EventCode +
            "座席：" +
            EventData +
            " 分机：" +
            login_ext +
            " 状态：" +
            EventInfo
        );
        console.log(callinfo);
        break;
      case 3: //3：表示外线状态发生变化，EventData 表示外线号码
        strStatus = "";
        show_trunk_status(document, EventData, strStatus);

        break;
      case 4: //4：表示分机状态发生变化，EventData 表示分机号
        strStatus = "";
        show_ext_status(document, EventData, strStatus);
        break;

      case 5: //5：表示其它座席传来消息，EventData 表示消息内容
        Popup_Callin(EventCode, EventData);
        break;

      case 6: //6：表示有用户呼入信息：应用程序依靠此消息弹出用户资料，EventData 表示分机号
        Popup_Callin(EventCode, EventData, EventInfo);
        break;

      case 7: //7：表示有OEM呼入信息，可能是FAX、VMS、EMAIL、SMS、CHAT等
        Popup_Callin(EventCode, EventData, EventInfo);
        break;
      case 8: //8：表示有多媒体会话状态发生变化，可能是SMS、WBC、WXC等，EventData 表示座席员工号，EventInfo：GID={0};CALLID={1};TYPE={2};STATUS={3};
        break;

      case 10: //10：表示呼叫进入排队，EventData 表示呼叫标识
        rem("呼叫进入排队" + EventData);
        break;
      case 11: //11：表示呼叫结束排队，EventData 表示呼叫标识
        rem("呼叫结束排队" + EventData);
        break;

      case 100: //100：表示与服务器的连接中断
        if (isConnect_CTI > 0) {
          isConnect_CTI = -1;
          check_cti_reconnect_1min = 0;
          var strDisp = "本座席与服务器的连接中断,原因：" + EventData;
          rem(strDisp, 0);
        }
        break;
    }
  }

  function get_call_status(strStatus, EventData) {
    var lbl_name;
    switch (strStatus) {
      case "01": //空闲
        lbl_name = "空闲";
        break;
      case "02": //应答
        lbl_name = "摘机";
        break;
      case "03": //振铃
        lbl_name = "振铃";
        window.callthinkglobal.ringEvent(EventData);
        break;
      case "04": //回铃
        lbl_name = "回铃";
        break;
      case "05": // 连接
        lbl_name = "通话";
        window.callthinkglobal.talkingEvent(EventData);
        break;
      case "06": //断开
        lbl_name = "断开";
        window.callthinkglobal.hangupEvent(EventData);
        break;
      case "07": //摘机
        lbl_name = "占用";
        break;
      case "08": //转移
        lbl_name = "转移";
        break;
      default: //未知
        lbl_name = "未知";
        break;
    }
    return lbl_name;
  }
```

```javascript
        url: O.path("call-center/api/get-config"),
        type: "get",
        callback: function (err, apiConfigData) {
          if (err) {
            publicTools.tips(err);
          } else {
            apiType = apiConfigData.api_type;
            apiConfig = apiConfigData.api_config;
            callback(apiType, apiConfig);
          }
        },
      });
    }
  });
  
```



## 开发环境
由于客服400是php后端渲染项目，目前开发环境不太完善。但是针对此400接入专项，我们可以手动配置一个开发环境。

一、准备一个静态资源服务器，比如nginx或者http-server都行 。

```plain
npm install --global http-server
```

二、将本地运行的main-resource.html的地址给到svip后端同学，然后客服业务拉取此地址运行。这样就可以在客服测试环境运行svip的本地代码了。
