```javascript
function curry (fn) {
    if (fn.length <= 1) return fn;
    const generator = function (...args1) {
        if (args1.length === fn.length) {
            return fn(...args1);
        } else {
            return (...args2) => {
                return generator(...args1,...args2)
            }
        }
```

```html
        let timer = null;
```

```html
        function throttle (fn,wait,{start,end} = {start:false,end:false}){
            let timer = null;
            let previous = 0
            const throttled = function (...args){
                const now = +new Date();
                // 第一次不立即执行的画面，更新判断事件
                if(!start && !previous){
                    previous = now;
                }
                // 处理之前延执行的定时器，进入有定时器的话，清空定时器；
                timer && clearTimeout(timer);
                // 判断点击是否满足事件要求；满足则触发事件，更新判断事件
                if (now - previous > wait) {
                    fn.apply(this,args);
                    previous = now;
                } else if (end){
                //  最后一次为延时执行时，则创建定时器；
                    timer = setTimeout(() => {
                        fn.apply(this,args);
                        previous = now;
                        timer = null
                    },wait)
                }
            }
            return throttled
        }
        let thr = throttle(fn,1000,{start:false,end:true})
        let oInput = document.getElementsByTagName('button')
        console.log(oInput)
        oInput[0].addEventListener('click',thr,false)
        
        function fn (){
            console.log('节流')
        }
    </script>
```

```javascript
function deepClone1 (target) {
    let data = null;
    if (Array.isArray(target)) {
        data = [];
    } else if (typeof target === 'object' && target !== null) {
        data = {};
    } else {
        return target;
    }

    for (let k in target) {
        if (target.hasOwnProperty(k)) {
            if ((typeof target[k] === 'object' && target[k] !== null) || Array.isArray(target)) {

                data[k] = target[k]

}
```

#### 2、栈
```javascript
function deepClone2 (target) {
    const root = {};
    const loopList = [
        {
            parent: root,
            key: undefined,
            data: target
        }
    ]
    while (loopList.length) {
        const {parent, key, data} = loopList.pop();
        let res = root;
        if (key !== undefined) {
            res = parent[key] = {};
        }
        for (let k in data) {
            if (data.hasOwnProperty(k)) {
                if (typeof data[k] === 'object' && data[k] !== null) {
                    loopList.push({
                        parent: res,
                        key: k,
                        data: data[k]
                    })
                } else {
                    res[k] = data[k]
                }
            }
        }
    }
    return root;
}
```

```javascript
const getData = function (url) {
    return new Promise((resolve,reject) => {
        const xhr = new XMLHttpRequest();
        xhr.open('GET', url);
        xhr.setRequestHeader('Accept','application/json');
        xhr.onreadystatechange = function () {
            if (xhr.readyState !== 4) {
                return false;
            }
            if (xhr.status === 200) {
                resolve(xhr.responseText);
            } else {
                reject(new Error(xhr.responseText));
            }
        }
        xhr.send();
    })
}
```

### 十二、JSONP
```javascript
function _JSONP (
    url,
    callback,
    callbackKey,
    params = {}
) {
    JSONP.callbackID = JSONP.callbackID || 1;
    const cbID = JSONP.callbackID;
    JSONP.callbacks = JSONP.callbacks || [];
    JSONP.callbacks[cbID] = callback;
    params[callbackKey] = `JSONP.callbacks[${cbID}]`;
    const paramsString = Object.keys(params).map(key => `${key}=${params[key]}`).join('&');

    const script = document.createElement('script');
    script.setAttribute('src',`${url}?${paramsString}`);
    document.body.appendChild(script);
    JSONP.callbackID++;
}
```
