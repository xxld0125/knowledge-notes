8.  looseEqual 判断两个数据是否相等  

```javascript
// 判断两数组是否相同
function looseCompareArrays(a, b) {
  	// 数组长度是否相同
    if (a.length !== b.length)
        return false;
    let equal = true;
  	// 遍历两数组内对应元素是否相同
    for (let i = 0; equal && i < a.length; i++) {
        equal = looseEqual(a[i], b[i]);
    }
    return equal;
}

function looseEqual(a, b) {
  	// 判断a&b是否全等
    if (a === b)
        return true;
    let aValidType = isDate(a);
    let bValidType = isDate(b);
  	// 判断a&b是否满足至少其中一个为date对象
    if (aValidType || bValidType) {
      	// 全为date对象,通过getTime判断是否相等
        return aValidType && bValidType ? a.getTime() === b.getTime() : false;
    }
    aValidType = isArray(a);
    bValidType = isArray(b);
  	// 判断a&b是否满足至少其中一个为数组对象
    if (aValidType || bValidType) {
      	// 全为date对象,通过looseCompareArrays判断是否相等
        return aValidType && bValidType ? looseCompareArrays(a, b) : false;
    }
    aValidType = isObject(a);
    bValidType = isObject(b);
  // 判断a&b是否满足至少其中一个为对象
    if (aValidType || bValidType) {
        /* istanbul ignore if: this if will probably never be called */
        if (!aValidType || !bValidType) {
            return false;
        }
        const aKeysCount = Object.keys(a).length;
        const bKeysCount = Object.keys(b).length;
      	// 判断两个对象key数量是否相等
        if (aKeysCount !== bKeysCount) {
            return false;
        }
      	// 遍历对象属性,判断各属性是否相同
        for (const key in a) {
            const aHasKey = a.hasOwnProperty(key);
            const bHasKey = b.hasOwnProperty(key);
            if ((aHasKey && !bHasKey) ||
                (!aHasKey && bHasKey) ||
                !looseEqual(a[key], b[key])) {
                return false;
            }
        }
    }
  	// 转字符串类型判断是否相同
    return String(a) === String(b);
}
```



9.  isSSRSafeAttrName 判断属性名是否包含特殊字符  

```javascript
// 特殊字符匹配正则
const unsafeAttrCharRE = /[>/="'\u0009\u000a\u000c\u0020]/;

// 缓存已校验过名称
const attrValidationCache = {};
function isSSRSafeAttrName(name) {
  	// 如果已校验过,直接返回上次校验结果
    if (attrValidationCache.hasOwnProperty(name)) {
        return attrValidationCache[name];
    }
    const isUnsafe = unsafeAttrCharRE.test(name);
    if (isUnsafe) {
        console.error(`unsafe attribute name: ${name}`);
    }
    return (attrValidationCache[name] = !isUnsafe);
}
```

```javascript
function stringifyStyle(styles) {
    let ret = '';
    if (!styles || isString(styles)) {
        return ret;
    }
    for (const key in styles) {
        const value = styles[key];
        const normalizedKey = key.startsWith(`--`) ? key : hyphenate(key);
        if (isString(value) ||
            (typeof value === 'number' && isNoUnitNumericStyleProp(normalizedKey))) {
            // only render valid values
            ret += `${normalizedKey}:${value};`;
        }
    }
    return ret;
}
```

11. normalizeClass  将class的数组、对象形式转化为字符串  

```javascript
function normalizeClass(value) {
    let res = '';
    if (isString(value)) {
        res = value;
    }
    else if (isArray(value)) {
        for (let i = 0; i < value.length; i++) {
            const normalized = normalizeClass(value[i]);
            if (normalized) {
                res += normalized + ' ';
            }
        }
    }
    else if (isObject(value)) {
        for (const name in value) {
            if (value[name]) {
                res += name + ' ';
            }
        }
    }
    return res.trim();
}
```

```javascript
function normalizeStyle(value) {
    if (isArray(value)) {
        const res = {};
        for (let i = 0; i < value.length; i++) {
            const item = value[i];
            const normalized = isString(item)
                ? parseStringStyle(item)
                : normalizeStyle(item);
            if (normalized) {
                for (const key in normalized) {
                    res[key] = normalized[key];
                }
            }
        }
        return res;
    }
    else if (isString(value)) {
        return value;
    }
    else if (isObject(value)) {
        return value;
    }
}
```

1.  makeMap  根据传入字符串,生成map,并返回一个检测指定key是否在该map中的函数

```javascript
/**
 * Make a map and return a function for checking if a key
 * is in that map.
 * IMPORTANT: all calls of this function must be prefixed with
 * \/\*#\_\_PURE\_\_\*\/
 * So that rollup can tree-shake them if necessary.

// 生成一个map并返回一个用来检测是否有指定key在该map中的函数
// 重要的是:所有调用的函数必须携带前缀,并满足:\/\*#\_\_PURE\_\_\*\/
// 这样rollup就可以在必要时进行tree-shake


function makeMap(str, expectsLowerCase) {
    const map = Object.create(null);
    const list = str.split(',');
    for (let i = 0; i < list.length; i++) {
        map[list[i]] = true;
    }
  	// expectsLowerCase 是否转小写后判断
    return expectsLowerCase ? val => !!map[val.toLowerCase()] : val => !!map[val];
}
```



2. cacheStringFunction 接受一个方法并返回一个函数,该函数接受字符串,之前未处理过该字符串时,调用传入的方法,并进行缓存,之前已处理过,则直接返回缓存结果;  

```javascript
const cacheStringFunction = (fn) => {
    const cache = Object.create(null);
    return ((str) => {
        const hit = cache[str];
        return hit || (cache[str] = fn(str));
    });
};
```
