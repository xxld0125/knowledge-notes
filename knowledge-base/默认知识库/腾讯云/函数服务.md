通过函数服务，转发open-ai域名下的接口。

```javascript
const express = require('express');
const httpProxy = require('http-proxy');
const { parse } = require('url');
const { format } = require('util');

const app = express();
const targetUrl = 'https://api.openai.com'; // 目标域名和端口
const target = new URL(targetUrl);
const proxy = httpProxy.createProxyServer({
  target: targetUrl,
  changeOrigin: true,
  secure: false,
});

// 修改请求头，将Host设置为目标域名
proxy.on('proxyReq', (proxyReq, req) => {
  proxyReq.setHeader('Host', target.host);
  proxyReq.setHeader('X-Forwarded-Host', target.host);
  proxyReq.setHeader('X-Forwarded-Proto', target.protocol);
});

// 打印HTTP请求和响应的日志
const logRequest = (req, res, next) => {
  const requestDump = formatRequest(req);
  console.log(`${getCurrentTimestamp()} Request: \n${requestDump}`);

  proxy.on('proxyRes', (proxyRes, req, res) => {
    const responseDump = formatResponse(proxyRes);
    console.log(`Response: \n${responseDump}`);
  });

  next();
};

const formatRequest = (req) => {
  const requestDump = format('%s %s HTTP/%s\n', req.method, req.originalUrl, req.httpVersion);

  const headers = Object.keys(req.headers)
    .map((key) => format('%s: %s', key, req.headers[key]))
    .join('\n');

  return `${requestDump}${headers}`;
};

const formatResponse = (res) => {
  const responseDump = format('HTTP/%s %d %s\n', res.httpVersion, res.statusCode, res.statusMessage);

  const headers = Object.keys(res.headers)
    .map((key) => format('%s: %s', key, res.headers[key]))
    .join('\n');

  return `${responseDump}${headers}`;
};

const getCurrentTimestamp = () => {
  return new Date().toISOString().replace(/T/, ' ').replace(/\..+/, '');
};

// 设置日志前缀和输出位置
console.logWithTimestamp = (message) => {
  console.log(`${getCurrentTimestamp()} ${message}`);
};

// 设置代理请求中间件
app.use((req, res, next) => {
  logRequest(req, res, next);
});

// 设置代理路由
app.all('*', (req, res) => {
  proxy.web(req, res);
});

// 启动服务器
const server = app.listen(9000, () => {
  console.logWithTimestamp('Starting server on port 9000...');
});

// 错误处理
proxy.on('error', (err, req, res) => {
  console.error('Proxy error:', err);
  res.status(500).send('Internal Server Error');
});

// 关闭服务器时关闭代理连接
server.on('close', () => {
  proxy.close();
});
```
