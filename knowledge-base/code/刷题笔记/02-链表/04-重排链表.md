### 题目一: [143. 重排链表](https://leetcode-cn.com/problems/reorder-list/)


#### 解法一: 
**思路: **

**1、找到中间节点**

**2、后半段的链表节点翻转**

**3、前段链表和后端链表进行拼接**

****

**代码**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {void} Do not return anything, modify head in-place instead.
 */
var reorderList = function(head) {

    if (!head || !head.next) {
        return head;
    }

    const dummy = new ListNode(null);
    dummy.next = head;

    half = halfNode(dummy);

  
    let right = half.next;
    half.next = null;
    let left = dummy.next;
    right = reverseList(right);


    while (left && right) {
        let tmp = left.next;
        let cur = right.next;

        right.next = left.next;
        left.next = right;

        left = tmp;
        right = cur;
    }
};

function halfNode(head) {

    let fast = head;
    let slow = head;

    while (fast && fast.next && fast.next) {
        fast = fast.next.next;
        slow = slow.next;
    }

    return slow;
}

function reverseList(head) {
    let cur = head;
    let prev = null;
    while (cur) {
        const tmp = cur.next;
        cur.next = prev;
        prev = cur;
        cur = tmp;
    }
    return prev;
}
```



### 题目二:  [1721. 交换链表中的节点](https://leetcode-cn.com/problems/swapping-nodes-in-a-linked-list/)
#### 解法一: 快慢指针
**思路: **

**通过快慢指针的方法, 找交换的两个节点;**



**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var swapNodes = function(head, k) {
    if (!head.next) {
        return head;
    }
    
    let tmp = new ListNode(-1, head);
    const res = tmp.next;
    
    // 通过快慢指针的方法, 找到交换的两个节点
    let i = 0, fast = slow = tmp;
    while (i < k) {
        i++;
        fast = fast.next;
    }
    const fastNode = fast;
    while (fast) {
        fast = fast.next;
        slow = slow.next;
    }

    // 交换值
    [fastNode.val, slow.val] = [slow.val, fastNode.val];

    return res;
};
```

#### 解法二: 
**思路:通过链表长度和给定的位置, 找到需要交换的两个节点, 再交换值**

**代码：**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var swapNodes = function(head, k) {
    if (!head.next) {
        return head;
    }
    
    let tmp = new ListNode(-1, head);
    const res = tmp.next;

    // 得到链表的长度
    let len = 0;
    let lenNode = head;
    while (lenNode) {
        len++;
        lenNode = lenNode.next;
    }

    // 找到需要交换的两个node
    let count = -1;
    let first, second;
    while (tmp) {
        count++;
        if (count === k) {
            first = tmp;
        }
        if (count === len + 1 - k ) {
            second = tmp;
        }
        tmp = tmp.next;
    }

    // 交换值
    [first.val, second.val] = [second.val, first.val];

    return res;
};
```
