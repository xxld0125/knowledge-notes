### 题目一: [86. 分隔链表](https://leetcode-cn.com/problems/partition-list/)
**思路: 循环链表, 将大于值和小于值的链表节点存入两个链表, 最后将两个链表按大小链接;**

**代码**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} x
 * @return {ListNode}
 */
var partition = function(head, x) {
    let min = new ListNode(null);
    let max = new ListNode(null);
    let tmp = max;
    let cur = min;

    while (head) {
        if (head.val < x) {
            min.next = head;
            min = min.next;
        } else {
            max.next = head;
            max = max.next;
        }
        head = head.next;
    }

    min.next = tmp.next;
    max.next = null;

    return cur.next;
};
```



### 题目二: [725. 分隔链表](https://leetcode-cn.com/problems/split-linked-list-in-parts/)
**思路:获取链表的长度, 根据给定的k, 计算出每个元素的长度.再拆分链表;**

**代码**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode[]}
 */
var splitListToParts = function(head, k) {
    // 获取链表的长度;
    let n = 0;
    let tmp = head;
    while (tmp) {
        n++;
        tmp = tmp.next;
    }
    // 获取每部分最小值, 和每部分都取最小值时的余值
    let partCount  = Math.floor(n / k), restCount = n % k;

    // 初始化输出值;
    const res = new Array(k).fill(null);

    let cur = head;

    for (let i = 0;i < k && cur;i++) {
        res[i] = cur;
        const  length = partCount +  (i < restCount ? 1 : 0);
        for (let m = 1;m < length;m++ ) {
            cur = cur.next;
        }
        const next = cur.next;
        cur.next = null;
        cur = next;
    }
    return res;
};
```



### 题目三:[817. 链表组件](https://leetcode-cn.com/problems/linked-list-components/)
**思路:循环迭代链表, 判断当前节点存在目标数组中,但前一个节点不在目标数组中,即组件数+1;**

**代码**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number[]} nums
 * @return {number}
 */
var numComponents = function(head, nums) {
    if (!head) return 0;

    let count = 0;

    let cur = new ListNode(null, head);
    let tmp = cur.next;
    let prev = cur;
    while (tmp) {
        if (nums.includes(tmp.val) && !nums.includes(prev.val)) {
            count++;
        }
        tmp = tmp.next;
        prev = prev.next;

    }
    return count;
};
```



### 题目四:[1669. 合并两个链表](https://leetcode-cn.com/problems/merge-in-between-linked-lists/)
**思路**: 找到指定删除下标a前面一个节点和b位置节点, 再进行拼接;

**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} list1
 * @param {number} a
 * @param {number} b
 * @param {ListNode} list2
 * @return {ListNode}
 */
var mergeInBetween = function(list1, a, b, list2) {
    let res = new ListNode(null, list1);
    let tmp = res;

    let list2H = list2;

    while (list2.next) {
        list2 = list2.next;
    }

    let start, end;

    let count = -1;

    while (tmp.next) {
        count++;
        if (count === a) {
            start = tmp
        }

        if (count === b) {
            end = tmp.next
        }
        
        tmp = tmp.next;
    }

    start.next = list2H;

    list2.next = end.next;

    return res.next

};
```
