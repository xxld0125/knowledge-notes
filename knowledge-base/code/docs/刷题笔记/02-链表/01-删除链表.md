var removeNthFromEnd = function(head, n) {
    let res = new ListNode(null, head);
    let slow = res;
    let fast = res;

    let s = n;
    while (s--) {
        fast = fast.next;
    }

    while (slow) {
        if (fast && fast.next === null) {
            slow.next = slow.next.next;
        }
         slow = slow.next;

        if (fast) {
            fast = fast.next;
        }
    }

    return res.next;
};

#### 解法二: 暴力解法
**思路:  先遍历链表, 获取链表的长度, 再遍历找到要删除节点的前一个节点, 进行删除操作**

**代码**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} n
 * @return {ListNode}
 */
var removeNthFromEnd = function(head, n) {
  	// 获取链表的长度
    let countList = head;
    let length = 1;
    while(countList) {
        length++;
        countList = countList.next;
    }
		
  	// 找到删除链表节点的前一个节点
    const target = length - n;
    let num = 1;
    let res = new ListNode(null, head);
    let tmp = res;

    while (tmp && tmp.next) {
        if (num === target) {
          	// 删除节点
            tmp.next = tmp.next.next;
            tmp = tmp.next;
        } else {
            tmp = tmp.next
        }
        num++;
    }
    return res.next;
};
```







### 题目二: [82. 删除排序链表中的重复元素 II](https://leetcode-cn.com/problems/remove-duplicates-from-sorted-list-ii/)


**思路**: 循环判断后续元素值是否相等,将前节点与后第一个不相等的节点进行链接;



**代码**:

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteDuplicates = function(head) {
    let res = new ListNode(null, head);
    let cur = res;
    while (cur && cur.next && cur.next.next) {
        if (cur.next.val === cur.next.next.val) {
            let tmp = cur.next.next;
            while (tmp && tmp.next && tmp.val === tmp.next.val) {
                tmp = tmp.next;
            }
            cur.next = tmp.next;
        } else {
            cur = cur.next
        }
    }
    return res.next;
};
```

### 题目三:[1171. 从链表中删去总和值为零的连续节点](https://leetcode-cn.com/problems/remove-zero-sum-consecutive-nodes-from-linked-list/)
#### 解法一: 前缀和
**思路: 第一次遍历,将链表节点根据计算得到的前缀和存入字典中, 此时相同前缀和的节点会进行覆盖.两个重复的节点中间即相加为0,在第二次遍历中,直接跳过该部分即可.**

**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var removeZeroSumSublists = function(head) {
    const map = new Map();
    const tmp = new ListNode(0, head);
    let curr = tmp;
    let sum = 0;
    while (curr) {
        sum += curr.val;
        map.set(sum, curr);
        curr = curr.next;
    }

    sum = 0;
    let curr2 = tmp;
    while (curr2){
        sum += curr2.val;
        curr2.next = map.get(sum).next;
        curr2 = curr2.next;
    }
    return tmp.next;
};
```

### 题目三:[2095. 删除链表的中间节点](https://leetcode-cn.com/problems/delete-the-middle-node-of-a-linked-list/)


#### 解法一:
**思路: 使用快慢指针找到中间节点的前一个节点, 然后删除中间节点**

**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteMiddle = function(head) {
    const res = new ListNode(null, head);

    let fast = res, slow = res;

    while (fast.next?.next) {
        fast = fast.next.next;
        slow = slow.next;
    }

    slow.next = slow.next.next;
    return res.next;
};
```

#### 解法二:
**思路: 遍历链表获取链表长度, 再找到中间链表的下标,再循环一次,删除中间节点;**

**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteMiddle = function(head) {
    // 获取链表长度
    let len = 0;
    let lenNode = head;
    const res = new ListNode(null, head);
    let tmp = res;
    while (lenNode) {
        lenNode = lenNode.next;
        len++;
    }
    
    // 找到需要删除的中间节点的下标
    const middle = len % 2 ? (len + 1) / 2 : len / 2 + 1;

    let count = 0;
    while (tmp) {
        count++;
        if (count === middle) {
            tmp.next = tmp.next.next;
        }
        tmp = tmp.next;
    }

    return res.next;


};
```

### 题目四:[2181. 合并零之间的节点](https://leetcode-cn.com/problems/merge-nodes-in-between-zeros/)
#### 解法一:
**思路: **题目要求将零节点之间的数字进行求和合并，那么可以直接采用遍历的方式：遇到零节点就创建一个新节点，并将后续的非零节点进行求和，之后再将这些新节点串起来。对于这种的方法，显然**需要额外的空间消耗**。



题目中提到“修改”以及“删除”，则可以不用创建新节点。那么怎么来处理呢？由于最后返回的零节点都不存在了，因此零节点在链表中充当的就是一个容器的功能，可以直接将后续非零节点的值在零节点上进行求和；至于如何串起来，则是遇零节点就将之前的零节点指向当前新的零节点，并更新先前的零节点指针。



这里比较特殊的一点是：如果是遇到最后一个零节点时（即cur.val==0 && !cur.next），不能再串了，而是直接置空，否则最后仍会保留一个零节点。



**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var mergeNodes = function(head) {
    // 方法二:
    let cur = head, prev = null;

    while (cur) {
        if (cur.val === 0) {
            if (prev) {
                prev.next = cur.next ? cur : null;
            }
            prev = cur;
        } else {
            prev.val += cur.val;
        }
        cur = cur.next;
    }
    return head;
};
```

#### 解法二:
**思路: 将链表节点存入数组中, 利用数组方法进行操作**

**代码:**

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var mergeNodes = function(head) {    
    // 将链表节点存入数组
    const list = [];
    while (head) {
        list.push(head);
        head = head.next;
    }
    
    // 将对应区域内的节点存入同一个数组元素
    let res = [];
    let count = -1;
    for (let i = 0;i < list.length;i++) {
        if (list[i].val === 0) {
            count++;
        } else {
            if (!res[count]) {
                res[count] = [];
            }
            res[count].push(list[i].val)
        }
    }
  
    // 统一区域内的节点树进行相加
    res = res.map(v => {
        return v.reduce((prev, cur) => {
            return prev + cur;
        }, 0)
    })

    // 将计算后的数组生成新的链表
    let result = new ListNode(0);
    let tmp = result;
    for (let i = 0;i < res.length;i++) {
        tmp.next = new ListNode(res[i]);
        tmp = tmp.next;
    }
    tmp.next = null;
    return result.next;
};
```
