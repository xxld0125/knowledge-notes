#### 一、创建本地数据库
离线APP采用的是将数据存储在本地的sqlite数据库中，那么本地的sqlite数据库中的表是如何创建的?

在代码中，我们会维护代码中使用到的表的创建sql文件 `init.sql`。

```sql
CREATE TABLE IF NOT EXISTS my_table
(
  id                  TEXT PRIMARY KEY NOT NULL COLLATE NOCASE,
  name                TEXT NOT NULL,
  created_on          DATETIME DEFAULT (datetime('now', 'localtime')),
  modified_on         DATETIME DEFAULT (datetime('now', 'localtime')),
  is_deleted          INTEGER DEFAULT 0,
);
```



在APP初始化时，会根据该sql文件，创建本地的db表。

```javascript
import sqlContent from '../sql/init.sql';
import db from './db';

function init() {
  // ...
  let content = sqlContent;
  sqls = content.split(';');
  if (sqls.length) {
    let promise = Promise.resolve();

    sqls.forEach(sql => {
      // 过滤掉多余的换行符，防止sql执行出错
      sql = sql.replace(/^[\n\r]$/, '');
      let isEmpty = sql.trim() === '';
      if (sql && sql.length && !isEmpty) {
        promise = promise.then(() => {
          return db.executeSql(sql);
        })
      }
    })
  }
  // ...
}
```



#### 二、db热更新
web热更新只更新web源码，但APP的功能运行还需要依赖于本地的数据库，而在新版中，数据库里的某些表结构是有可能会发生变化的，当数据库表结构发生变化时，init.sql 肯定需要进行做相应修改，但仅仅在这里修改是不够的，因为 init.sql 只是用来首次创建数据库时才会使用到，简单说，也就是只用来在APP开始使用前创建好各种空表。那么会出现的场景有两种：

+ 用户新安装应用
+ 用户使用APP的清除离线数据功能

那么，最常见的场景，用户已经用了APP，而且程序进行了web热更新，代码都是运行的最新版的了，在这种场景下，如果新版有数据表结构变化的需求时，在程序更新到新版时，如果让本地数据库也进行相应的升级呢？

这个过程，称为db热更新，执行时机在于web热更新之后重新加载首页的初始化过程中。

数据库的版本升级管理过程需要严格记录每个版本的升级处理，因为是会出现跨版本升级场景的，那么就需要将从原版本升级到目标版本这之间所有的数据库升级处理都执行一遍，这样数据库表才会更新到最新。

离线APP的数据库版本升级管理是结合前端和后端一起实现的，APP只负责db热更新检测处理，当需要进行热更新时，需要的升级sql则交由后端维护负责。

以下是相关代码：

```javascript
import db from './db';

/**
 * 执行列更新
 *
 * @param {string} sqls sql语句集合
 * @param {Object} existsObj 需要判断存在的集合（表名和列名）
 * @param {Function} addFinishCount 更新完成sql的数量的方法
 * @returns
 */
let _execColumn = (sqls, existsObj, addFinishCount) => {
  return db.existsColumn(existsObj.table_name, existsObj.column_name).then(exists => {
    if (existsObj.exists === exists) {
      // 不存在列时才执行的sql, 必须满足 1.存在表,2.不存在列
      if (!existsObj.exists) {
        return db.existsTable(existsObj.table_name).then(exists => {
          if (exists) {
            return _execSql(sqls, addFinishCount);
          }
        });
      }

      return _execSql(sqls, addFinishCount);
    } else {
      addFinishCount(sqls.length);
    }
  });
};

/**
 * 执行表更新的sql
 *
 * @param {Array} sqls sql语句集合
 * @param {Object} existsObj 需要判断存在的集合（表名）
 * @param {Function} addFinishCount 更新完成sql的数量的方法
 * @returns
 */
let _execTable = (sqls, existsObj, addFinishCount) => {
  return db.existsTable(existsObj.table_name).then(exists => {
    if (exists === existsObj.exists) {
      return _execSql(sqls, addFinishCount);
    } else {
      addFinishCount(sqls.length);
    }
  });
};

/**
 * 获取升级数据库sql, sql热更新
 *
 * @param {string} appCode app的编号
 * @param {string} currentVersion 当前版本号
 * @param {string} targetVersion 目标版本号
 * @returns
 */
let _getUpdateDbSQL = (appCode, currentVersion, targetVersion) => {
  return http
    .getData(
      `${getApiUrl('app_tools_site')}/api/update/app-version/get-db-script`,
      {
        appCode: appCode,
        currentDBVersion: encodeURIComponent(currentVersion),
        targetDBVersion: encodeURIComponent(targetVersion),
      },
      'get'
    )
    .then(result => {
      if (result.errCode != 0) {
        return Promise.reject(result.errMsg);
      }
      return Promise.resolve(result.data);
    })
    .catch(error => {
      return Promise.reject(error);
    });
};

/**
 * 更新数据库
 *
 * @param {string} appCode APP的编号
 * @param {string} currentVersion 当前版本号
 * @param {string} targetVersion 目标版本号
 * @returns
 */
let _updateDb = (appCode, currentVersion, targetVersion) => {
  let finishCount = 0; // 执行完成的数量
  let promise = Promise.resolve();

  let addFinishCount = n => {
    finishCount += n || 1;
  };

  // 获取升级的sql
  return (_updatePromise = _getUpdateDbSQL(appCode, currentVersion, targetVersion).then(sqls => {
    if (!sqls.length) {
      return Promise.resolve('暂无升级脚本');
    }

    sqls.forEach(sqlObj => {
      if ('exec_if_exists_column' in sqlObj) {
        // 2.1.处理存在字段才执行的sql
        sqlObj.exec_if_exists_column.exists = true;

        promise = promise.then(() => {
          return _execColumn(sqlObj.sql, sqlObj.exec_if_exists_column, addFinishCount);
        });
      } else if ('exec_if_not_exists_column' in sqlObj) {
        // 2.2.处理不存在字段才执行的sql
        sqlObj.exec_if_not_exists_column.exists = false;

        promise = promise.then(() => {
          return _execColumn(sqlObj.sql, sqlObj.exec_if_not_exists_column, addFinishCount);
        });
      } else if ('exec_if_not_exists_table' in sqlObj) {
        // 2.3 处理不存在表才执行的sql
        sqlObj.exec_if_not_exists_table.exists = false;
        promise = promise.then(() => {
          return _execTable(sqlObj.sql, sqlObj.exec_if_not_exists_table, addFinishCount);
        });
      } else if ('exec_if_exists_table' in sqlObj) {
        // 2.4 处理存在表才执行的sql
        sqlObj.exec_if_exists_table.exists = true;
        promise = promise.then(() => {
          return _execTable(sqlObj.sql, sqlObj.exec_if_exists_table, addFinishCount);
        });
      } else {
        // 2.3.处理纯sql
        promise = promise.then(() => {
          return _execSql(sqlObj.sql, addFinishCount);
        });
      }
    });

    return promise.then(() => {
      return Promise.resolve(finishCount);
    });
  }));
};
```



#### 三、不使用db热更新实现表结构更新
上面介绍的是通过接口获取更新sql，这种方法需要后端配合，且更新sql需要另外管理。其实也可以讲更新的sql放在仓库代码中维护，这样可以避免后端配合，且无需另外管理。

`update_sql.js`文件的热更新依赖web热更新即可。

```javascript
import updateSql from '../sql/update_sql';
import db from './db';

let _getUpdateDbSQL = () => {
  return Promise.resolve().then(() => {
    return updateSql;
  });
};

/**
 * 更新数据库
 * @param {string} appCode APP的编号
 * @param {string} currentVersion 起始版本号
 * @param {string} targetVersion 目标版本号
 * @returns
 */
let _updateDb = (appCode, currentVersion, targetVersion) => {
  let finishCount = 0; // 执行完成的数量
  let promise = Promise.resolve();

  let addFinishCount = n => {
    finishCount += n || 1;
  };

  // 获取升级的sql
  return (_updatePromise = _getUpdateDbSQL(currentVersion, targetVersion).then(sqls => {
    if (!sqls.length) {
      return Promise.resolve('暂无升级脚本');
    }

    sqls.forEach(sqlObj => {
      if ('exec_if_exists_column' in sqlObj) {
        // 2.1.处理存在字段才执行的sql
        sqlObj.exec_if_exists_column.exists = true;

        promise = promise.then(() => {
          return _execColumn(sqlObj.sql, sqlObj.exec_if_exists_column, addFinishCount);
        });
      } else if ('exec_if_not_exists_column' in sqlObj) {
        // 2.2.处理不存在字段才执行的sql
        sqlObj.exec_if_not_exists_column.exists = false;

        promise = promise.then(() => {
          return _execColumn(sqlObj.sql, sqlObj.exec_if_not_exists_column, addFinishCount);
        });
      } else if ('exec_if_not_exists_table' in sqlObj) {
        // 2.3 处理不存在表才执行的sql
        sqlObj.exec_if_not_exists_table.exists = false;
        promise = promise.then(() => {
          return _execTable(sqlObj.sql, sqlObj.exec_if_not_exists_table, addFinishCount);
        });
      } else if ('exec_if_exists_table' in sqlObj) {
        // 2.4 处理存在表才执行的sql
        sqlObj.exec_if_exists_table.exists = true;
        promise = promise.then(() => {
          return _execTable(sqlObj.sql, sqlObj.exec_if_exists_table, addFinishCount);
        });
      } else {
        // 2.3.处理纯sql
        promise = promise.then(() => {
          return _execSql(sqlObj.sql, addFinishCount);
        });
      }
    });

    return promise.then(() => {
      return Promise.resolve(finishCount);
    });
  }));
};
```



这种方法相较管理简单，但是存在一个问题，该方案的db热更新依赖web热更新。当db热更新的需要灰度发布时，必须和web热更新的一起发布。但之前的方法db热更新和web热更新可以分开管理，灵活性更高。
