### 一、PC端
#### 1、[vue-upload-component](https://lian-yue.github.io/vue-upload-component/#/zh-cn/documents)
PC上传依赖[vue-upload-component](https://lian-yue.github.io/vue-upload-component/#/zh-cn/documents), 根据业务场景对其进行封装。



#### 2、上传地址
一般是前端直传阿里云oss, 或者调取接口，存在自己公司的服务器。

**oss配置封装示例**

```javascript
// 获取oss配置
import { fetchStsConfig } from '@/api/commonApi';
import Vue from 'vue';

// 对上传minio 和 alioss封装处理的插件 
import WebUpload from '@yl/web-upload';

let instance = null;
const OSS = require('ali-oss');

// 使用 iframe 下载
function downloadByIframe(src) {
  const iframeId = 'iframeDownloadId';
  let iframe = document.getElementById(iframeId);
  if (!iframe) {
    iframe = document.createElement('iframe');
    iframe.id = iframeId;
    iframe.style.width = '0px';
    iframe.style.height = '0px';
    iframe.style.overflow = 'hidden';
  }
  iframe.src = src;
  document.body.appendChild(iframe);
}

export const init = () => {
  return fetchStsConfig()
    .then(res => {
      const result = res.data;
      instance = new OSS({
        accessKeyId: result.AccessKeyId,
        accessKeySecret: result.AccessKeySecret,
        stsToken: result.SecurityToken,
        endpoint: result.Endpoint,
        bucket: result.Bucket
      });
      instance.altInfo = {
        customDomain: result.CustomDomain
      };
      return res.data;
    })
    .catch(err => {
      console.log(err);
      Vue.prototype.$message({
        message: '获取stsconfig失败',
        type: 'warning'
      });
    });
};

export const getOSSInstance = () => {
  return init().then(() => {
    return Promise.resolve(instance);
  });
};

export const multiPartUpload = (fileName, fileObj, options, stsConfig) => {
  return fetchStsConfig().then(res => {
    if (stsConfig) {
      console.error('[upload-oss.js]/stsConfig', stsConfig);
    }
    const result = stsConfig || res.data;
    //该插件filObj只支持file格式文件
    return WebUpload.doUpload({
      accessKeyId: result.AccessKeyId,
      secretAccessKey: result.AccessKeySecret,
      sessionToken: result.SecurityToken,
      endpoint: result.Endpoint,
      bucket: result.Bucket,
      expiration: result.Expiration,
      key: fileName,
      fileObj,
      ossType: result.Type,
      ...options
    }).then(r => {
      return { ...r, customDomain: result.CustomDomain };
    });
  });
};

export const cancel = WebUpload.cancel;

// 获取能下载的文件网络地址
export const getDownloadUrl = (originUrl, key) => {
  return fetchStsConfig()
    .then(res => {
      const result = res.data;
      const url = WebUpload.downloadFile({
        key: originUrl.replace(result.CustomDomain + '/', ''),
        accessKeyId: result.AccessKeyId,
        secretAccessKey: result.AccessKeySecret,
        sessionToken: result.SecurityToken,
        endpoint: result.Endpoint,
        bucket: result.Bucket,
        expiration: result.Expiration,
        ossType: result.Type,
        fileName: key
      });
      return url;
    })
    .then(res => {
      console.log(res, '下载');
      return res;
    });
};

export const downloadFileByPrivate = (url, key) => {
  return getDownloadUrl(url, key).then(res => {
    downloadByIframe(res);
  });
};

```



#### 3、上传组件封装示例
```vue
<template>
  <div class="upload-wrapper">
    <!--上传按钮-->
    <vue-upload
      v-show="!!maxNum"
      ref="uploader"
      v-model="files"
      class="btn btn-primary uploader"
      :class="{ disabled: !!maxNum }"
      :input-id="inputId"
      :extensions="filters.extensions"
      :accept="filters.accept"
      :size="maxSize * sizeUnit"
      :multiple="multiple"
      :maximum="maxNum"
      @input-filter="inputFilter"
      @input="changeFile"
    >
      <div class="btn-wrapper">
        <div class="btn"><i class="ic iconfont">&#xe71a;</i>{{ buttonText }}</div>
        <span v-if="customTips !== ''" class="upload-tip">{{ customTips }}</span>
        <span v-else class="upload-tip"
          >({{ maxNum > 20 ? '' : `已上传${showList.length}/${maxNum}个文件，` }}单个文件最大支持{{ maxSize }}M，支持{{
            filters.extensions.split(',').join('/')
          }}文件格式)</span
        >
      </div>
    </vue-upload>
    <div v-if="showList.length" class="list">
      <ul v-if="showImage" class="img-list clearfix">
        <template v-for="(item, i) in showList">
          <template v-if="['jpg', 'jpeg', 'png', 'gif'].includes(item.suffix)">
            <!--显示图片-->
            <li v-if="!item.notEnd" :key="'img' + i">
              <el-image
                :ref="`img${i}`"
                class="pic"
                :src="item.file_url"
                fit="cover"
                :preview-src-list="
                  showList.filter(v => ['jpg', 'jpeg', 'png', 'gif'].includes(v.suffix)).map(v => v.file_url)
                "
              ></el-image>
              <div class="popover">
                <div class="icon-container">
                  <i class="el-icon-zoom-in" @click="previewImg(i)"></i>
                  <i class="el-icon-delete" @click="delItem(i)"></i>
                </div>
              </div>
            </li>
            <!--图片加载状态-->
            <li v-else :key="'img' + i" class="circle">
              <el-progress type="circle" :width="64" :percentage="parseInt(item.p * 100)"></el-progress>
              <div class="popover">
                <div class="icon-container">
                  <i class="el-icon-delete" @click="cancelUpload(i)"></i>
                </div>
              </div>
            </li>
          </template>
        </template>
      </ul>

      <template v-for="(item, i) in showList">
        <div
          v-if="item && (!showImage || !['jpg', 'jpeg', 'png', 'gif'].includes(item.suffix))"
          :key="i"
          class="list-item"
        >
          <i class="doc-icon" :class="createFileIconCls(item.suffix)"></i>
          <template v-if="!item.notEnd">
            <div class="list-item-filename">{{ item.file_name || item.name }}</div>
            <div class="list-item-option-btn">
              <a v-if="filePreviewHost" class="list-item-del" @click="previewFile(item)">预览</a>
              <a v-if="isShowDowload" href="javascript:void(0);" class="list-item-del" @click="download(item)">下载</a>
              <a class="list-item-del" href="javascript:;" @click="delItem(i)">删除</a>
            </div>
          </template>
          <div v-else class="list-item-ipt progress">
            <div class="progressName">
              <div class="file-name">{{ item.name }}</div>
              <div class="file-delete" @click="cancelUpload(i)">&times;</div>
            </div>
            <div class="progressLine">
              <div class="color-line" :style="{ width: parseInt(item.p * 100) + '%' }"></div>
            </div>
          </div>
        </div>
      </template>
    </div>
  </div>
</template>
<script>
import VueUpload from 'vue-upload-component';
import { multiPartUpload, cancel, downloadFileByPrivate, getDownloadUrl } from '@/utils/upload-oss';

export default {
  name: 'OssUpload',
  components: {
    VueUpload
  },
  model: {
    prop: 'fileList',
    event: 'change'
  },
  props: {
    fileList: {
      type: Array,
      default: () => []
    },
    inputId: {
      type: String,
      default: () => new Date().getTime() + '_' + Math.floor(Math.random(1) * 1000)
    },
    businessId: {
      type: String,
      default: 'checkfile'
    },
    required: {
      type: Boolean,
      default: false // 上传文件的前提条件
    },
    reUpload: {
      type: Boolean,
      default: false
    },
    requiredTip: {
      type: String,
      default: ''
    },
    maxSize: {
      type: Number,
      default: 500
    },
    maxNum: {
      type: Number,
      default: 500
    },
    buttonText: {
      type: String,
      default: '添加资料'
    },
    filters: {
      type: Object,
      default: () => ({
        extensions: 'pptx,ppt,xlsx,xls,docx,doc,pdf,jpg,jpeg,png,gif',
        accept: '*'
      })
    },
    // 变更的文件列表，用于提交数据给后端
    handleFileList: {
      type: Object,
      default: () => {
        return {
          add: [],
          delete: []
        };
      }
    },
    defaultFieldConfig: {
      type: Object,
      default: () => {}
    },
    // 图片文件是否以缩略图展示
    showImage: {
      type: Boolean,
      default: false
    },
    // 支持外部手动修改路径
    convertPathFn: {
      type: Function,
      default: null
    },
    // 支持外部手动设置 sts config
    stsConfig: {
      type: Object,
      default: null
    },
    // 自定义提示文字
    customTips: {
      type: String,
      default: ''
    },
    // 是否支持下载
    isShowDowload: {
      type: Boolean,
      default: true
    }
  },
  data() {
    return {
      text: '',
      files: [],
      sizeUnit: 1024 * 1024,
      showList: [],
      deleteIdx: [],
      isUploading: false,
      currType: '', // 标记上传
      fieldConfig: {
        fileName: 'file_name',
        disabled: 'disabled',
        suffix: 'suffix',
        fileSize: 'file_size',
        fileUrl: 'file_url'
      },
      filePreviewHost: ''
    };
  },
  computed: {
    // maxNum 大于 1 时，才允许选择多张
    multiple() {
      return this.maxNum > 1;
    }
  },
  watch: {
    defaultFieldConfig: {
      handler(newV) {
        Object.assign(this.fieldConfig, newV || {});
      },
      immediate: true
    },
    fileList: {
      handler(val) {
        if (val) {
          this.showList = this.fileList.map(v => {
            if (typeof v === 'string') {
              return v;
            }
            const name = v[this.fieldConfig.fileName] || '';
            let suffix = v[this.fieldConfig.suffix] || '';
            if (!suffix && name.indexOf('.') > -1) {
              suffix = name.substr(name.lastIndexOf('.') + 1);
            }
            return {
              file_name: name,
              disabled: v[this.fieldConfig.disabled] || false,
              suffix: suffix,
              file_size: v[this.fieldConfig.fileSize] || 0,
              file_url: v[this.fieldConfig.fileUrl] || '',
              ...v
            };
          });
        }
      },
      immediate: true
    }
  },
  created() {
    this.getFilePreviewHost();
  },
  methods: {
    getTenantCode() {
      const url = location.href;
  
      const origin = getOrigin();
  
      // 去除 query 参数
      const realPath = url.split('?')[0];
      // 去除域名，端口，query 参数
      const finalPath = realPath.slice(origin.length);
  
      // 以 / 作为标志，获取一个结果集
      const result = finalPath.match(/([^\/?]+)/g) || [];
  
      const code = result.length ? result[0] : '';
      // IE9 有 # 号
      // if (code === '#') {
  
      // }
  
      return code;
    },
    getOssFileName (tenantCode, businessId, suffix) {
      const _s4 = () => {
        return (((1 + Math.random()) * 0x10000) | 0).toString(16).substring(1);
      };
      const uuid = () => {
        return `${_s4()}${_s4()}-${_s4()}-${_s4()}-${_s4()}-${_s4()}${_s4()}${_s4()}`;
      };
      const year = new Date().getFullYear();
      const month = String(new Date().getMonth() + 1).padStart(2, '0');
      const day = String(new Date().getDate()).padStart(2, '0');
      // 使用生成的 uuid 作为文件名，原生文件名里有带中文的场景，这种场景生成的 url 存在兼容性问题
      return `${tenantCode}/${businessId}/${year}/${month}/${day}/${uuid()}.${suffix}`;
    },
    createFileIconCls(fileType) {
      var type = '';
      switch (fileType) {
        case 'jpg':
        case 'jpeg':
        case 'png':
        case 'gif':
          type = 'img';
          break;
        case 'vsd':
          type = 'vsd';
          break;
        case 'vsdx':
          type = 'vsd';
          break;
        case 'pptx':
        case 'ppt':
          type = 'ppt';
          break;
        case 'xlsx':
        case 'xls':
          type = 'xls';
          break;
        case 'docx':
        case 'doc':
          type = 'doc';
          break;
        case 'pdf':
          type = 'pdf';
          break;
        default:
          break;
      }
      return type;
    },
    formatFileList() {
      const arr = this.showList.map(v => {
        return this.formatFile(v);
      });
      this.$emit('change', arr);
    },
    formatFile(v) {
      const item = {
        [this.fieldConfig.disabled]: !!v.disabled,
        [this.fieldConfig.fileName]: v.file_name,
        [this.fieldConfig.suffix]: v.suffix,
        [this.fieldConfig.fileSize]: v.file_size ? v.file_size : parseInt(v.size / 1024),
        [this.fieldConfig.fileUrl]: v.file_url
      };
      if (v.id) {
        item.id = v.id;
      }
      return item;
    },
    // 删除文件
    delItem(idx) {
      this.showList.splice(idx, 1);
      const [file] = this.fileList.splice(idx, 1);
      if (file.id) {
        if (this.handleFileList) {
          this.handleFileList.delete = this.handleFileList.delete || [];
          this.handleFileList.delete.push(file);
          this.$emit('update:handleFileList', {
            ...this.handleFileList
          });
        }
      } else if (this.handleFileList && this.handleFileList.add) {
        const findIndex = this.handleFileList.add.findIndex(
          v => v[this.fieldConfig.fileUrl] === file[this.fieldConfig.fileUrl]
        );
        if (findIndex > -1) {
          this.handleFileList.add.splice(findIndex, 1);
          this.$emit('update:handleFileList', {
            ...this.handleFileList
          });
        }
      }
    },
    inputFilter(newFile, oldFile, prevent, type = '', reUpload = false) {
      this.currType = type;
      // 上传文件的前提条件
      if (this.required) {
        this.$message({
          message: this.requiredTip,
          type: 'warning'
        });
        return prevent();
      }
      const curNum = this.showList.length + 1;

      if (curNum > this.maxNum) {
        this.$message({
          message: `最多只能上传${this.maxNum}个文件`,
          type: 'warning'
        });
        return prevent();
      }
      // 如果超过 maxNum。当 maxNum 为一张时，允许替换
      if (this.maxNum === 0 && !reUpload) {
        this.$message({
          message: `最多只能上传${this.showList.length}个文件`,
          type: 'warning'
        });

        // 清空
        const uploader = this.$refs['uploader' + this.currType].clear
          ? this.$refs['uploader' + this.currType]
          : this.$refs['uploader' + this.currType][0];
        uploader.clear();
        return prevent();
      }

      if (newFile && !oldFile) {
        let errMsg = '';

        // 过滤文件格式, IE9 没有type, 使用 name 匹配
        const type = `${newFile.name.split(/\./).pop()}`;
        if (this.filters.extensions.indexOf(type) < 0) {
          errMsg = '请按格式要求上传文件';
        }

        // 如果图片size超过限制
        if (newFile.size > this.maxSize * this.sizeUnit) {
          errMsg = '文件大小已超出限制';
        }

        if (errMsg) {
          this.$message({
            message: errMsg,
            type: 'warning'
          });
          return prevent();
        }
      }
    },
    // 文件上传状态变更
    changeList(data, type) {
      if (this.deleteIdx.indexOf(data.key) > -1) {
        cancel();
        return;
      }
      if (type === 'progress') {
        this.$set(this.showList, data.key, data);
      }
      if (type === 'uploaded') {
        // 上传结束
        this.$set(this.showList, data.key, {
          fileMold: {
            id: '',
            name: '请选择类型'
          },
          fileMoldList: this.docTypeList,
          name: data.file_name,
          file_name: data.file_name,
          ...data
        });
        const file = this.showList[data.key];
        console.log('uploaded', file);
        if (this.handleFileList && this.handleFileList.add) {
          this.handleFileList.add.push(this.formatFile(file));
          this.$emit('update:handleFileList', {
            ...this.handleFileList
          });
        }
      }
    },
    cancelUpload(idx) {
      this.deleteIdx.push(idx);
      this.$set(this.showList, idx, null);
    },
    allUploaded() {
      this.showList = this.showList.filter(v => !!v);
      this.deleteIdx = [];
      const uploader = this.$refs['uploader' + this.currType].clear
        ? this.$refs['uploader' + this.currType]
        : this.$refs['uploader' + this.currType][0];
      uploader.clear();
      this.formatFileList();
    },
    previewImg(file) {
      this.$refs['img' + file][0].clickHandler();
    },
    changeFile(files, item = {}, index = -1) {
      if (this.showList.length + files.length > this.maxNum) {
        return this.$message({
          message: `最多只能上传${this.maxNum}个文件`,
          type: 'warning'
        });
      }
      const promiseArr = [];
      const tenantCode = this.getTenantCode();
      if (this.isUploading) {
        this.$message({
          message: '还有正在上传的任务，请稍后',
          type: 'warning'
        });
        return;
      }
      files.forEach((file, key) => {
        key = index > -1 ? index : this.showList.length + key;
        this.isUploading = true;
        this.$emit('progress', {
          p: 0,
          name: file.name,
          file_name: file.name,
          notEnd: true,
          key,
          suffix: file.name.split('.')[1]
        });
        const vm = this;
        const promise = new Promise((resolve, reject) => {
          const [fileName, fileSuffix] = file.name.split('.');
          const coverPath = this.convertPathFn && this.convertPathFn(fileName, fileSuffix);
          const path = coverPath || this.getOssFileName(tenantCode, this.businessId, fileSuffix, fileName);
          multiPartUpload(
            path,
            file.file,
            {
              progress: p => {
                console.log('上传分片:', p);
                vm.changeList(
                  {
                    p,
                    // ...checkpoint,
                    name: file.name,
                    file_name: item.file_name || file.name,
                    notEnd: true,
                    key,
                    suffix: file.name.split('.')[1]
                  },
                  'progress'
                );
              }
            },
            this.stsConfig
          )
            .then(result => {
              vm.changeList(
                {
                  // file_id: item.id,
                  name: file.name,
                  file_name: item.file_name || file.name,
                  scopes: item.scopes || [],
                  room_count_detail: item.room_count_detail || [],
                  suffix: file.name.split('.').pop(),
                  file_path: path,
                  size: file.size,
                  file_url: result.customDomain + '/' + path,
                  key
                },
                'uploaded'
              );
              resolve();
            })
            .catch(err => {
              if (err.name !== 'cancel' && err !== '上传失败') {
                vm.$message({
                  message: file.name + ' 上传失败',
                  type: 'warning'
                });
              }
              return resolve();
            });
        });

        promiseArr.push(promise);
      });
      Promise.all(promiseArr).then(() => {
        this.isUploading = false;
        this.allUploaded();
      });
    },
    download(item) {
      const urlKey = this.fieldConfig.fileUrl;
      const nameKey = this.fieldConfig.fileName;
      downloadFileByPrivate(item.downLoadUrl || item.url || item[urlKey], item.fileName || item.name || item[nameKey]);
    },
    async previewFile(item) {
      const urlKey = this.fieldConfig.fileUrl;
      const nameKey = this.fieldConfig.fileName;
      const url = await getDownloadUrl(
        item.downLoadUrl || item.url || item[urlKey],
        item.fileName || item.name || item[nameKey]
      );
      window.open(`${this.filePreviewHost}?url=${encodeURIComponent(url)}`, '_blank');
    },
    getFilePreviewHost() {
      if (this.filePreviewHost) {
        return;
      }
      if (this.$store && this.$store.state.app && this.$store.state.app.filePreviewHost) {
        this.filePreviewHost = this.$store.state.app.filePreviewHost;
      }
      if (!this.filePreviewHost) {
        this.filePreviewHost = localStorage.getItem('filePreviewHost');
      }
    }
  }
};
</script>
```

```javascript
const TAG = '[web-upload]';

    from: ''

let lastStsConfig = null;

const _s4 = () => {
  return (((1 + Math.random()) * 0x10000) | 0).toString(16).substring(1);
};

function uuid() {
  return `${_s4()}${_s4()}-${_s4()}-${_s4()}-${_s4()}-${_s4()}${_s4()}${_s4()}`;
}

/**
 * 根据文件名获取文件后缀
 * @param {*} name
 * @returns
 */
function getSuffix(name) {
  if (name && name.lastIndexOf('.') > -1) {
    return name.substr(name.lastIndexOf('.') + 1);
  }
  return '';
}

function init(options) {
  Object.assign(defaultOptions, options);
  console.log(TAG, '初始化 => 前端直传', defaultOptions);
}

async function fetchStsConfig() {
  const now = new Date().getTime();
  if (lastStsConfig && now - lastFetchStsConfigTime < 5000) {
    //  5s 内的上传，直接复用之前的配置
    return lastStsConfig;
  } else {
    lastStsConfig = null;
  }
  const res = await http.get(defaultOptions.stsConfigUrl, defaultOptions.stsParams);
  if (res?.errcode === 0) {
    lastFetchStsConfigTime = now;
    lastStsConfig = res.data;
    return res.data;
  }
}

/**
 * 获取文件上传的路径
 * @param {*} foler：文件夹，fileName：文件名
 * @returns
 */
export function getUploadFilePath({ folder = 'default', fileName = '' }) {
  const year = new Date().getFullYear();
  const month = String(new Date().getMonth() + 1).padStart(2, '0');
  const day = String(new Date().getDate()).padStart(2, '0');
  const suffix = getSuffix(fileName);
  // 使用生成的 uuid 作为文件名，原生文件名里有带中文的场景，这种场景生成的 url 存在兼容性问题
  return `${folder}/${year}/${month}/${day}/${uuid()}${suffix ? '.' + suffix : ''}`;
}

    console.log(TAG, 'fetchStsConfig()', config?.Type);

    sessionToken: config.SecurityToken,

    expiration: config.Expiration,
    key: filePath,
    fileObj,
    ossType: config.Type,
    ...options
  };
  let res = null;

    res = await WebUpload.doUpload(_options);
```

```javascript
  if (url) {

/**

function getSuffix(name) {
  if (name && name.lastIndexOf('.') > -1) {

/**
 * 将文件大小（单位：byte）转换为带单位的字符串
 * @param {*} size
 * @returns
 */
function sizeTostr(size) {
  var data = '';
  size = Number(size);
  if (size < 1 * 1024) {
    //如果小于1KB转化成B
    data = size.toFixed(2) + 'B';
  } else if (size < 1 * 1024 * 1024) {
    //如果小于1MB转化成KB
    data = (size / 1024).toFixed(2) + 'KB';
  } else if (size < 1 * 1024 * 1024 * 1024) {
    //如果小于1GB转化成MB
    data = (size / (1024 * 1024)).toFixed(2) + 'MB';
  } else {
    //其他转化成GB
    data = (size / (1024 * 1024 * 1024)).toFixed(2) + 'GB';
  }
  var sizestr = data + '';
  var len = sizestr.indexOf('.');
  var dec = sizestr.substr(len + 1, 2);
  if (dec == '00') {
    //当小数点后为00时 去掉小数部分
    return sizestr.substring(0, len) + sizestr.substr(len + 3, 2);
  }
  return sizestr;
}

 * 文件上传组件的逻辑，抽离与 UI 无关的逻辑，方便不同端复用

export default class FileUploader {
  fileList;

  options = {
    folder: 'default', // oss 文件上传的模块目录名
    maxCount: 5, // 默认最多上传 5 份
    maxSize: 20971520, // 默认最大上传 20M
    toastFn: () => {}, // 弹窗提示的方法，由外部注入
    filters: []   // 过滤文件格式
  };

  // 字段映射配置
  defaultField = {
    name: 'name',
    suffix: 'suffix',
    size: 'size',
    url: 'url'
  };

  constructor(options) {
    Object.assign(this.options, options);
  }

  updateFileList(fileList) {
    if (fileList) {
      fileList.forEach(v => this.parseFileInfo(v));
    }
    this.fileList = fileList;
    console.log(TAG, 'updateFileList', this.fileList);
  }

  /**
   * 更新字段映射配置
   * @param {*} fieldConfig
   */
  updateFieldConfig(fieldConfig) {
    Object.assign(this.defaultField, fieldConfig);
    console.log(TAG, 'updateFieldConfig', this.defaultField);
  }

  /**
   * 更新文件上传的相关配置
   * @param {*} options
   */
  updateOptions(options) {
    Object.assign(this.options, options);
    console.log(TAG, 'updateOptions', this.options);
  }

  /**
   * 预处理文件信息，添加上文件相关信息字段
   * @param {*} file
   */
  parseFileInfo(file) {
    const url = file[this.defaultField.url] || '';
    file[this.defaultField.url] = url;
    const name = file[this.defaultField.name] || file.file?.name || getFileName(url) || '';
    file[this.defaultField.name] = name;
    file[this.defaultField.suffix] = getSuffix(name);
    file[this.defaultField.size] = file[this.defaultField.size] || file.file?.size || '';
  }

  /**
   * 选择完文件，但在文件被放入待上传列表前执行
   * @param {Object | Array} file 选完文件返回的 File 对象数据，可以是一个文件对象，也可以是文件数组对象
   * @returns true 表示可以上传，false 表示拦截此次选择的文件上传
   */
  beforeRead(file) {
    let files = file;
    if (!Array.isArray(file)) {
      files = [file];
    }
    console.log(TAG, 'beforeRead()', files);
    const maxCount = this.options.maxCount;
    if (maxCount > 0 && this.fileList.length + files.length > maxCount) {
      const message = '文件数量不能超过' + maxCount;
      this.options.toastFn({ message: message, duration: 5000 });
      return false;
    }
    // 限制上传文件格式
    const suffix = getSuffix(file.name).toLowerCase();
    if (!this.options.filters.includes(suffix)) {
      this.options.toastFn({ message: '请按照要求的格式上传文件', duration: 5000 });
      return false;
    }
    return true;
  }

  /**
   * 选择完文件，并通过预处理，加入待上传列表后执行，可在这里执行上传文件逻辑
   * @param {Object | Array} file 包裹了原生 File 数据的 wrapper 对象，增加了文件上传相关信息字段，e.g {file: File, status: '', message: ''}，可以是一个文件对象，也可以是数组对象
   * @param {*} detail
   */
  async afterRead(file, detail) {
    console.log(TAG, 'afterRead()', detail, file);
    let files = file;
    if (!Array.isArray(file)) {
      files = [file];
    }
    files.forEach(file => {
      this.parseFileInfo(file);
    });
    const { folder } = this.options;
    await Promise.all(
      files.map(async file => {
        file.status = 'uploading';
        file.message = '上传中';
        const filePath = getUploadFilePath({ folder: folder, fileName: file.file.name });
        try {
          const res = await doUpload(filePath, file.file, {
            progress: p => {
              file.message = '上传中(' + (p * 100).toFixed(0) + '%)';
            }
          });
          file.status = 'done';
          file.url = res.url;
          file.message = '';
          // 埋点日志
          window.$log('文件上传', {
            ...file,
            filePath: filePath
          });
        } catch (error) {
          console.error(TAG, file.file.name, error);
          // 埋点日志
          window.$log('文件上传', {
            ...file,
            filePath: filePath
          });
          this.options.toastFn(file.file.name + '上传失败');
          file.status = 'failed';
          file.message = '上传失败';
        }
      })
    );
    console.log(TAG, 'all file upload', this.fileList);
  }

  /**
   * 文件超出限制大小时调用，传入超出大小的文件
   * @param {*} file {Object | Array} file 包裹了原生 File 数据的 wrapper 对象，增加了文件上传相关信息字段，e.g {file: File, status: '', message: ''}，可以是一个文件对象，也可以是数组对象
   * @param {*} detail
   */
  onOversize(file, detail) {
    let files = file;
    if (!Array.isArray(file)) {
      files = [file];
    }
    const message = files.map(v => v.file?.name || '').join('，') + '文件不能超过' + sizeTostr(this.options.maxSize);
    console.log(TAG, 'onOversize()', message, detail, file);
    this.options.toastFn({ message: message, duration: 5000 });
  }
}
```

一般建议是直传oss。但是wxsdk提供`getImgBase64Data`处理`chooseImage`获取的本地图片，返回的是base64格式的文件，但是`ali-oss`不支持。前端处理的话，需要先将`base64`转 `blob`对象 转 `arrayBuffer`对象 转 `buffer`对象，或者通过后端进行oss直传。

以上两种方法的话，建议采用调取后端接口的方式直传, 主要是前端转换格式的话，如果文件过大，且本地的服务器性能过差的话，需要花费较长的时间，不稳定。

```javascript
const calSize = (base64url) => {
  let str = base64url.replace('data:image/png;base64,', '');
  const equalIndex = str.indexOf('=');
  if (str.indexOf('=') > 0) {
    str = str.substring(0, equalIndex);
  }
  const strLength = str.length;
  const fileLength = strLength - (strLength / 8) * 2;
  // 返回单位为MB的大小
  return fileLength;
};

const getImgBase64Data = (localId) => {
  return new Promise((resolve) => {
    getWx().ready(() => {
      getWx().getLocalImgData({
        localId: localId, // 图片的localID
        success: function (localImg) {
          let localData = localImg.localData; // localData是图片的base64数据，可以用img标签显示
          resolve({
            url: localData,
            file: localData,
            size: calSize(localData),
          });
        },
      });
    });
  });
};

  getImgBase64Data,
```

```vue
function guid() {

  }
  return (

    S4() +
```
