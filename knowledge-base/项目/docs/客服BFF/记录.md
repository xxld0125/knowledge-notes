#### 二、环境配置
##### 1、`grpc-code-gen.config.js`配置
```javascript
const {
  default: Config, grpcResolvePath, getServiceConfig,
  deserializeGrpcStatusDetails, ErrorPb,
} = require('@yunke/yungen');

module.exports = Config((config) => {

  // gitUrls: 写相应的微服务proto仓库地址（若不知可咨询相关的后台开发人员）
  config.gitUrls = [
    {
      url: 'git@git.myscrm.cn:ykcommon/ykproto.git',
      branch: 'master',
    },
  ];

  // clientOptions: 主要处理 grpc 请求的证书配置, 当服务名为 antman-banner时配置为antman,当服务名为uc-user-center是配置为uc, 其他的同理
  config.clientOptions = (defaultOptions) => {
    const exec = /^(antman|thanos|uc)-.+/.exec(defaultOptions['grpc.ssl_target_name_override']);
    if (exec) {
      defaultOptions['grpc.ssl_target_name_override'] = exec[1];
    }
    return defaultOptions;
  };

  return config;
});
```



#### 2、`grpc-service.dev.config.js`配置
```javascript
module.exports = {
  "antman-banner": { // 微服务名称 （根据自身业务决定）
    "host": "112.74.75.140", // grpc请求host地址，开发环境默认为 112.74.75.140
    "port": 58487, // 请求相应微服务的端口号
    "defaultCa": true // 是否使用默认的证书，开发环境统一配置为 true 使用默认证书配置
  },
};

```
